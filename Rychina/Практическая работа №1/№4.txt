#include <iostream>
using namespace std;

int main()
{
    setlocale(LC_ALL, "Russian");
    int arr[10],n, k = 0;
  
    cout << "Введите массив:";
    for (int i = 0; i < 10; i++)
        
        cin >> arr[i];

    for (int i = 0; i < 10 - 1; i++) {
        for (int j = 0; j < 10 - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {

                n = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = n;
            }
        }
    }

    for (int i = 0; i < 10 - 1; i++)
    {
        if (arr[i] == arr[i + 1]) k++;
        else { cout << arr[i] << "  " << k + 1 << "\n"; k = 0; }
        if (i == 10 - 2) cout << arr[i + 1] << "  " << k + 1 << "\n";
    }
    
    for (int j = 0; j < 10; j++)
    {
        cout << arr[j]<< " ";
    }

    return 0;
}

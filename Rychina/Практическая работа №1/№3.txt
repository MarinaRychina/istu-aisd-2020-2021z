#include <iostream>
using namespace std;
int Recursion(int arr[100], int n)

{

    if (n <= 0)

        return 0;

    return (Recursion(arr, n - 1) + arr[n - 1]);

}

int main()

{
    setlocale(LC_ALL, "Russian");
    int n, i;
    cout << "Введите количество элементов массива:";
    cin >> n;
    int arr[100];
    cout << "Введите  массив:";
    for (i = 0; i < n; i++)
        cin >> arr[i];

    cout << Recursion(arr, n);

    return 0;

}
